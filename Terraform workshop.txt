# contains terraform settings
terraform {
  required_version = ">= 0.13"
  required_providers {
    aws = {
      # shorthand for registry, terraform.io/hashicorp/aws
      source = "hashicorp/aws"
      version = "~>3.0"
    }
  }
}

# block configures the aws provider 
provider "aws" {
  region = "us-east-1"
  profile = "default"
}
